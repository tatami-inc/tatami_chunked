82 void set(Index_ i) {
83 selection = OracularSubsettedSlabCacheSelectionType::BLOCK;
84 block_start = i;
85 block_end = i + 1;
86 indices.clear();
87 mapping.clear();
88 }
89
90 void add(Index_ i) {
91 if (selection == OracularSubsettedSlabCacheSelectionType::FULL) {
92 return;
93 }
94
95 if (selection == OracularSubsettedSlabCacheSelectionType::BLOCK) {
96 if (i == block_end) {
97 block_end = i + 1;
98 return;
99
100 } else if (i + 1 == block_start) {
101 block_start = i;
102 return;
103
104 } else if (i >= block_start && i < block_end) {
105 return;
106 }
107
108 selection = OracularSubsettedSlabCacheSelectionType::INDEX;
109 indices.resize(block_end - block_start);
110 std::iota(indices.begin(), indices.end(), block_start);
111 fill_mapping();
112 }
113
114 if (mapping.find(i) == mapping.end()) {
115 mapping[i] = indices.size();
116 indices.push_back(i);
117 }
118 }
119
120 void finalize() {
121 if (selection == OracularSubsettedSlabCacheSelectionType::BLOCK) {
122 block_length = block_end - block_start;
123 } else if (selection == OracularSubsettedSlabCacheSelectionType::INDEX) {
124 if (!std::is_sorted(indices.begin(), indices.end())) {
125 std::sort(indices.begin(), indices.end());
126 fill_mapping();
127 }
128 }

349 // all_slabs should be referenced by a pointer in future_cache.
350 // There shouldn't be any free cache entries remaining in

354 // worrying about leaking memory. The only exception is if we're at
