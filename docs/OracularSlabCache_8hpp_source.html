140 Index_ index = this->next();
141 auto slab_info = identify(index);
142 if (slab_info.first == my_last_slab_id && my_last_slab) {
143 return std::make_pair(my_last_slab, slab_info.second);
144 }
145 my_last_slab_id = slab_info.first;
146
147 // Updating the cache if we hit the refresh point.
148 if (my_counter - 1 == my_refresh_point) {
149 // Note that, for any given populate cycle, the first prediction's
150 // slab cannot already be in the cache, otherwise it would have
151 // incorporated into the previous cycle. So we can skip some code.
152 my_future_cache[slab_info.first] = NULL;
153 my_in_need.push_back(slab_info.first);
154 size_t used_slabs = 1;
155 auto last_future_slab_id = slab_info.first;
156
157 while (++my_refresh_point < my_total) {
158 auto future_index = my_oracle->get(my_refresh_point);
159 auto future_slab_info = identify(future_index);
160 if (last_future_slab_id == future_slab_info.first) {
161 continue;
162 }
163
164 last_future_slab_id = future_slab_info.first;
165 if (my_future_cache.find(future_slab_info.first) != my_future_cache.end()) {
166 continue;
167 }
168
169 if (used_slabs == my_max_slabs) {
170 break;
171 }
172 ++used_slabs;
173
174 auto ccIt = my_current_cache.find(future_slab_info.first);
175 if (ccIt == my_current_cache.end()) {
176 my_future_cache[future_slab_info.first] = NULL;
177 my_in_need.push_back(future_slab_info.first);
178
179 } else {
180 auto slab_ptr = ccIt->second;
181 my_future_cache[future_slab_info.first] = slab_ptr;
182 my_current_cache.erase(ccIt);
183 if constexpr(track_reuse_) {
184 my_to_reuse.emplace_back(future_slab_info.first, slab_ptr);
185 }
186 }
187 }
188
189 auto cIt = my_current_cache.begin();
190 for (auto a : my_in_need) {
191 if (cIt != my_current_cache.end()) {
192 my_to_populate.emplace_back(a, cIt->second);
193 my_future_cache[a] = cIt->second;
194 ++cIt;
195 } else {
196 // We reserved my_all_slabs so further push_backs() should not
197 // trigger any reallocation or invalidation of the pointers.
198 my_all_slabs.push_back(create());
199 auto slab_ptr = &(my_all_slabs.back());
200 my_to_populate.emplace_back(a, slab_ptr);
201 my_future_cache[a] = slab_ptr;
202 }
203 }
204 my_in_need.clear();
205
206 if constexpr(track_reuse_) {
207 populate(my_to_populate, my_to_reuse);
208 } else {
209 populate(my_to_populate);
210 }
211
212 my_to_populate.clear();
213 if constexpr(track_reuse_) {
214 my_to_reuse.clear();
215 }
216
217 // We always fill my_future_cache to the brim so every entry of
218 // my_all_slabs should be referenced by a pointer in
219 // my_future_cache. There shouldn't be any free cache entries
220 // remaining in my_current_cache i.e., at this point, cIt should
221 // equal my_current_cache.end(), as we transferred everything to
222 // my_future_cache. Thus it is safe to clear my_current_cache
223 // without worrying about leaking memory. The only exception is if
224 // we run out of predictions, in which case it doesn't matter.
225 my_current_cache.clear();
226 my_current_cache.swap(my_future_cache);
227 }
228
229 // We know it must exist, so no need to check ccIt's validity.
230 auto ccIt = my_current_cache.find(slab_info.first);
231 my_last_slab = ccIt->second;
232 return std::make_pair(my_last_slab, slab_info.second);
