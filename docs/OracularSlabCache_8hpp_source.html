131 my_last_slab_id = slab_info.first;
132
133 // Updating the cache if we hit the refresh point.
134 if (my_counter - 1 == my_refresh_point) {
135 // Note that, for any given populate cycle, the first prediction's
136 // slab cannot already be in the cache, otherwise it would have
137 // incorporated into the previous cycle. So we can skip some code.
138 my_future_cache[slab_info.first] = NULL;
139 my_in_need.push_back(slab_info.first);
140 size_t used_slabs = 1;
141 auto last_future_slab_id = slab_info.first;
142
143 while (++my_refresh_point < my_total) {
144 auto future_index = my_oracle->get(my_refresh_point);
145 auto future_slab_info = identify(future_index);
146 if (last_future_slab_id == future_slab_info.first) {
147 continue;
148 }
149
150 last_future_slab_id = future_slab_info.first;
151 if (my_future_cache.find(future_slab_info.first) != my_future_cache.end()) {
152 continue;
153 }
154
155 if (used_slabs == my_max_slabs) {
156 break;
157 }
158 ++used_slabs;
159
160 auto ccIt = my_current_cache.find(future_slab_info.first);
161 if (ccIt != my_current_cache.end()) {
162 auto slab_ptr = ccIt->second;
163 my_future_cache[future_slab_info.first] = slab_ptr;
164 my_current_cache.erase(ccIt);
165 } else {
166 my_future_cache[future_slab_info.first] = NULL;
167 my_in_need.push_back(future_slab_info.first);
168 }
169 }
170
171 auto cIt = my_current_cache.begin();
172 for (auto a : my_in_need) {
173 if (cIt != my_current_cache.end()) {
174 my_to_populate.emplace_back(a, cIt->second);
175 my_future_cache[a] = cIt->second;
176 ++cIt;
177 } else {
178 // We reserved my_all_slabs so further push_backs() should not
179 // trigger any reallocation or invalidation of the pointers.
180 my_all_slabs.push_back(create());
181 auto slab_ptr = &(my_all_slabs.back());
182 my_to_populate.emplace_back(a, slab_ptr);
183 my_future_cache[a] = slab_ptr;
184 }
185 }
186 my_in_need.clear();
187
188 populate(my_to_populate);
189 my_to_populate.clear();
190
191 // We always fill my_future_cache to the brim so every entry of
192 // my_all_slabs should be referenced by a pointer in
193 // my_future_cache. There shouldn't be any free cache entries
194 // remaining in my_current_cache i.e., at this point, cIt should
195 // equal my_current_cache.end(), as we transferred everything to
196 // my_future_cache. Thus it is safe to clear my_current_cache
197 // without worrying about leaking memory. The only exception is if
198 // we run out of predictions, in which case it doesn't matter.
199 my_current_cache.clear();
200 my_current_cache.swap(my_future_cache);
201 }

204 auto ccIt = my_current_cache.find(slab_info.first);
205 my_last_slab = ccIt->second;
