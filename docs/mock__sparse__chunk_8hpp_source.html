62 static void refine_start_and_end(size_t& start, size_t& end, Index_ desired_start, Index_ desired_end, Index_ max_end, const std::vector<index_type>& indices) {
63 if (desired_start) {
64 auto it = indices.begin();
65 start = std::lower_bound(it + start, it + end, static_cast<index_type>(desired_start)) - it;
66 }
67
68 if (desired_end != max_end) {

70 if (start != end && indices[start] == desired_start) {
71 end = start + 1;
72 } else {
73 end = start;
74 }

101 template<bool is_block_, typename Index_>
102 void fill_target(
103 Index_ p,
104 Index_ non_target_start,
105 Index_ non_target_end,
106 Index_ non_target_chunkdim,
107 Workspace<value_type, index_type>& work,
108 const std::vector<value_type*>& output_values,
109 const std::vector<index_type*>& output_indices,
110 Index_* output_number,
111 index_type shift)
112 const {
113 size_t start = work.indptrs[p], end = work.indptrs[p + 1];
114 if (start >= end) {
115 return;
116 }
117
118 refine_start_and_end(start, end, non_target_start, non_target_end, non_target_chunkdim, work.indices);
119
120 auto& current_number = output_number[p];
121 const bool needs_value = !output_values.empty();
122 auto vptr = needs_value ? output_values[p] + current_number : NULL;
123 const bool needs_index = !output_indices.empty();
124 auto iptr = needs_index ? output_indices[p] + current_number : NULL;
125
126 if constexpr(is_block_) {
127 if (needs_value) {
128 std::copy(work.values.begin() + start, work.values.begin() + end, vptr);
129 }
130 if (needs_index) {
131 for (size_t i = start; i < end; ++i, ++iptr) {
132 *iptr = work.indices[i] + shift;
133 }
134 }
135 current_number += end - start;
136
137 } else {
138 // Assumes that work.remap has been properly configured, see configure_remap().
139 for (size_t i = start; i < end; ++i) {
140 Index_ target = work.indices[i];
141 if (work.remap[target]) {
142 if (needs_value) {
143 *vptr = work.values[i];
144 ++vptr;
145 }
146 if (needs_index) {
147 *iptr = target + shift;
148 ++iptr;
149 }
150 ++current_number;
151 }
152 }
153 }
