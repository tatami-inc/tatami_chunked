64 static void refine_start_and_end(size_t& start, size_t& end, Index_ desired_start, Index_ desired_end, Index_ max_end, const std::vector<index_type>& indices) {
65 if (desired_start) {
66 auto it = indices.begin();
67 start = std::lower_bound(it + start, it + end, static_cast<index_type>(desired_start)) - it;
68 }
69
70 if (desired_end != max_end) {
71 if (desired_end == desired_start + 1) {
72 if (start != end && indices[start] == desired_start) {
73 end = start + 1;
74 } else {
75 end = start;
76 }
77 } else {
78 auto it = indices.begin();
79 end = std::lower_bound(it + start, it + end, static_cast<index_type>(desired_end)) - it;
80 }

103 template<bool is_block_, typename Index_>
104 void fill_primary(
105 Index_ p,
106 Index_ secondary_start,
107 Index_ secondary_end,
108 Index_ secondary_chunkdim,
109 Workspace<value_type, index_type>& work,
110 std::vector<value_type>& current_values,
111 std::vector<index_type>& current_indices,
112 index_type shift)
113 const {
114 auto start = work.indptrs[p], end = work.indptrs[p + 1];
115 if (start >= end) {
116 return;
117 }
118
119 refine_start_and_end(start, end, secondary_start, secondary_end, secondary_chunkdim, work.indices);
120
121 if constexpr(is_block_) {
122 current_values.insert(current_values.end(), work.values.begin() + start, work.values.begin() + end);
123 for (size_t i = start; i < end; ++i) {
124 current_indices.push_back(work.indices[i] + shift);
125 }
126 } else {
127 // Assumes that work.remap has been properly configured, see configure_remap().
128 for (size_t i = start; i < end; ++i) {
129 Index_ target = work.indices[i];
130 if (work.remap[target]) {
131 current_values.push_back(work.values[i]);
132 current_indices.push_back(target + shift);
133 }
134 }
135 }
