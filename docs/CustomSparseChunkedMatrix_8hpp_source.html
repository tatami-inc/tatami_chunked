473template<class Slab_, typename Index_, typename Value_>
474tatami::SparseRange<Value_, Index_> process_sparse_slab(const std::pair<const Slab_*, Index_>& fetched, Value_* value_buffer, Index_* index_buffer, bool needs_value, bool needs_index) {
475 auto num = fetched.first->number[fetched.second];
476
477 if (needs_value) {
478 auto vptr = fetched.first->values[fetched.second];
479 std::copy_n(vptr, num, value_buffer);
480 } else {
481 value_buffer = NULL;
482 }
483
484 if (needs_index) {
485 auto iptr = fetched.first->indices[fetched.second];
486 std::copy_n(iptr, num, index_buffer);
487 } else {
488 index_buffer = NULL;
489 }
490
491 return tatami::SparseRange<Value_, Index_>(num, value_buffer, index_buffer);
492}

862 auto stats = [&]{
863 if (row) {

875 return SlabCacheStats(
876 my_coordinator.get_chunk_ncol(),
877 non_target_length,
878 my_coordinator.get_num_chunks_per_row(),
879 my_cache_size_in_bytes,
880 element_size,
881 my_require_minimum_cache
882 );
883 }
884 }();

886 auto wrk = my_manager->new_workspace_exact();
887 if (stats.max_slabs_in_cache == 0) {
888 return std::make_unique<Extractor_<true, oracle_, false, Value_, Index_, ChunkValue_, decltype(wrk)> >(std::move(wrk), my_coordinator, stats, row, std::forward<Args_>(args)...);
889 } else if constexpr(oracle_) {
890 if (my_cache_subset) {
891 return std::make_unique<Extractor_<false, true, true, Value_, Index_, ChunkValue_, decltype(wrk)> >(std::move(wrk), my_coordinator, stats, row, std::forward<Args_>(args)...);
892 } else {
