57 SparseSlabFactory<Index_, typename Chunk_::value_type, typename Chunk_::index_type> factory;

61 // holds the content for a single chunk, while final_solo holds the content
62 // across chunks but only for the requested dimension element. Both cases

64 // memory-efficient than trying to use num_slabs_in_cache = 1.

66 Slab final_solo;

68public:

73 Index_ secondary_length,
74 bool needs_value,
75 bool needs_index) :
76 coordinator(coordinator),
77 oracle(std::move(ora)),
78 factory(1, secondary_length, 1, needs_value, needs_index),
79 tmp_solo(
80 coordinator.template get_primary_chunkdim<accrow_>(),
81 coordinator.template get_secondary_chunkdim<accrow_>(),
82 needs_value,
83 needs_index
84 ),
85 final_solo(factory.create())
86 {}

90protected:

93 if constexpr(oracle_) {
94 i = oracle->get(counter++);

179template<class Slab_, typename Index_, typename Value_>
180tatami::SparseRange<Value_, Index_> process_sparse_slab(const std::pair<const Slab_*, Index_>& fetched, Value_* vbuffer, Index_* ibuffer, bool needs_value, bool needs_index) {
181 auto num = fetched.first->number[fetched.second];
182
183 if (needs_value) {
184 auto vptr = fetched.first->values[fetched.second];
185 std::copy_n(vptr, num, vbuffer);
186 } else {
187 vbuffer = NULL;
188 }
189
190 if (needs_index) {
191 auto iptr = fetched.first->indices[fetched.second];
192 std::copy_n(iptr, num, ibuffer);
193 } else {
194 ibuffer = NULL;
195 }
196
197 return tatami::SparseRange<Value_, Index_>(num, vbuffer, ibuffer);
198}

371struct DensifiedIndex : public tatami::DenseExtractor<oracle_, Value_, Index_>, public SparseBase<accrow_, solo_, oracle_, Value_, Index_, Chunk_> {
372 DensifiedIndex(

508 std::unique_ptr<Interface_<oracle_, Value_, Index_> > raw_internal(bool row, Index_ secondary_length, const tatami::Options& opt, Args_&& ... args) const {
509 size_t element_size = (opt.sparse_extract_value ? sizeof(typename Chunk_::value_type) : 0) + (opt.sparse_extract_index ? sizeof(typename Chunk_::index_type) : 0);
510

514 if (stats.num_slabs_in_cache > 0) {
515 return std::make_unique<Extractor_<true, false, oracle_, Value_, Index_, Chunk_> >(coordinator, stats.num_slabs_in_cache, std::forward<Args_>(args)...);
516 } else {
517 return std::make_unique<Extractor_<true, true, oracle_, Value_, Index_, Chunk_> >(coordinator, stats.num_slabs_in_cache, std::forward<Args_>(args)...);
518 }
519 } else {
520 // Remember, the num_chunks_per_row is the number of slabs needed to divide up all the *columns* of the matrix.
521 SlabCacheStats stats(coordinator.get_chunk_ncol(), secondary_length, coordinator.get_num_chunks_per_row(), cache_size_in_bytes, element_size, require_minimum_cache);
522 if (stats.num_slabs_in_cache > 0) {
523 return std::make_unique<Extractor_<false, false, oracle_, Value_, Index_, Chunk_> >(coordinator, stats.num_slabs_in_cache, std::forward<Args_>(args)...);
524 } else {
