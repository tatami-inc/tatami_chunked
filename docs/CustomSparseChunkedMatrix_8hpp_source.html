49protected:
50 const ChunkCoordinator<Index_, true, Chunk_, int>& coordinator;
51 typename Chunk_::Workspace chunk_workspace;
52 typedef typename ChunkCoordinator<Index_, true, Chunk_, int>::SparseSlab Slab;
53
54 tatami::MaybeOracle<oracle_, Index_> oracle;
55 typename std::conditional<oracle_, size_t, bool>::type counter = 0;

57 // These two instances are not fully allocated Slabs; rather, tmp_solo just
58 // holds the content for a single chunk, while final_solo holds the content
59 // across chunks but only for the requested dimension element. Both cases
60 // are likely to be much smaller than a full Slab, so we're already more

63 Slab final_solo;

65public:

75 {}

77 ~SparseBaseSolo() = default;

79protected:

82 if constexpr(oracle_) {
83 i = oracle->get(counter++);

156template<class Slab_, typename Index_, typename Value_>
157tatami::SparseRange<Value_, Index_> process_sparse_slab(const std::pair<const Slab_*, Index_>& fetched, Value_* vbuffer, Index_* ibuffer, bool needs_value, bool needs_index) {
158 const auto& values = fetched.first->values[fetched.second];
159 const auto& indices = fetched.first->indices[fetched.second];
160
161 if (needs_value) {
162 std::copy(values.begin(), values.end(), vbuffer);
163 } else {
164 vbuffer = NULL;
165 }
166
167 if (needs_index) {
168 std::copy(indices.begin(), indices.end(), ibuffer);
169 } else {
170 ibuffer = NULL;
171 }
172
173 return tatami::SparseRange<Value_, Index_>(values.size(), vbuffer, ibuffer);
174}
