101 void add(Index_ i) {
102 if (selection == SubsetSelection::FULL) {
103 return;
104 }
105
106 if (selection == SubsetSelection::BLOCK) {
107 if (i == block_end) {
108 block_end = i + 1;
109 return;
110
111 } else if (i + 1 == block_start) {
112 block_start = i;
113 return;
114
115 } else if (i >= block_start && i < block_end) {
116 return;
117 }
118
119 selection = SubsetSelection::INDEX;
120 indices.resize(block_end - block_start);
121 std::iota(indices.begin(), indices.end(), block_start);
122 fill_mapping();
123 }
124
125 if (mapping.find(i) == mapping.end()) {
126 mapping[i] = indices.size();
127 indices.push_back(i);
128 }
129 }
130
131 void finalize() {
132 if (selection == SubsetSelection::BLOCK) {
133 block_length = block_end - block_start;
134 } else if (selection == SubsetSelection::INDEX) {
135 if (!std::is_sorted(indices.begin(), indices.end())) {
136 std::sort(indices.begin(), indices.end());
137 fill_mapping();
138 }
139 }
140 }
