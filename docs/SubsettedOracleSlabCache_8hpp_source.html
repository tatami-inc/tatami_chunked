100 void add(Index_ i) {
101 if (selection == SubsetSelection::FULL) {
102 return;
103 }
104
105 if (selection == SubsetSelection::BLOCK) {
106 if (i == block_end) {
107 block_end = i + 1;
108 return;
109
110 } else if (i + 1 == block_start) {
111 block_start = i;
112 return;
113
114 } else if (i >= block_start && i < block_end) {
115 return;
116 }
117
118 selection = SubsetSelection::INDEX;
119 indices.resize(block_end - block_start);
120 std::iota(indices.begin(), indices.end(), block_start);
121 fill_mapping();
122 }
123
124 if (mapping.find(i) == mapping.end()) {
125 mapping[i] = indices.size();
126 indices.push_back(i);
127 }
128 }
129
130 void finalize() {
131 if (selection == SubsetSelection::BLOCK) {
132 block_length = block_end - block_start;
133 } else if (selection == SubsetSelection::INDEX) {
134 if (!std::is_sorted(indices.begin(), indices.end())) {
135 std::sort(indices.begin(), indices.end());
136 fill_mapping();
137 }
138 }
