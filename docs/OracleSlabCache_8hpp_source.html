128 // Updating the cache if we hit the refresh point.
129 if (counter - 1 == refresh_point) {

131 size_t used_slabs = 1;

134 while (++refresh_point < total) {

137 if (last_future_slab_id != future_slab_info.first) {
138 if (future_cache.find(future_slab_info.first) == future_cache.end()) {
139 if (used_slabs == max_slabs) {

143 ++used_slabs;
144 }
145 }
146 }

148 auto cIt = current_cache.begin();
149 for (auto a : to_reassign) {

151 future_cache[a] = cIt->second;

159 // We always fill future_cache to the brim so every entry of
160 // all_slabs should be referenced by a pointer in future_cache.
161 // There shouldn't be any free cache entries remaining in
162 // current_cache i.e., at this point, cIt should equal
163 // current_cache.end(), as we transferred everything to
164 // future_cache. Thus it is safe to clear current_cache without
165 // worrying about leaking memory. The only exception is if we're at
166 // the end of the predictions, in which case it doesn't matter.
167 current_cache.clear();

179 void requisition_slab(Id_ slab_id, Cfunction_ create) {
180 auto ccIt = current_cache.find(slab_id);
181 if (ccIt != current_cache.end()) {
182 auto slab_ptr = ccIt->second;
183 future_cache[slab_id] = slab_ptr;
184 current_cache.erase(ccIt);
185
186 } else if (all_slabs.size() < max_slabs) {
187 all_slabs.emplace_back(create());

189 future_cache[slab_id] = slab_ptr;
190 to_populate.emplace_back(slab_id, slab_ptr);
191
192 } else {
