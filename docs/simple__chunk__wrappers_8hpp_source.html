460 if (desired_start) {
461 auto it = indices.begin();
462 start = std::lower_bound(it + start, it + end, static_cast<typename SimpleChunk_::index_type>(desired_start)) - it;
463 }
464
465 if (desired_end != max_end) {
466 if (desired_end == desired_start + 1) {
467 if (start != end && indices[start] == desired_start) {
468 end = start + 1;
469 } else {
470 end = start;
471 }
472 } else {
473 auto it = indices.begin();
474 end = std::lower_bound(it + start, it + end, static_cast<typename SimpleChunk_::index_type>(desired_end)) - it;
